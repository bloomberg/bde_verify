function(llvmlib name)
    find_library(LIB${name}
                 NAMES ${name}
                 PATHS ${LLVM_LIBRARY_DIRS}
                       /lib64
                       /opt/bb/lib64
                       /opt/swt/lib64
                       /usr/lib/sparcv9
                       /usr/lib64
    )
    if(LIB${name})
        target_link_libraries(bde_verify_bin ${LIB${name}})
    endif()
endfunction(llvmlib)
